
// empty values: initial tests

$ 
> [ ["Value",""] ]

$ ;
> [ ["Value",""], ["",null] ]

$ ;;
> [ ["Value",""], ["",null], ["",null] ]

$ a
> [ ["Value", "a"] ]

$ aa
> [ ["Value", "aa"] ]

$ aa;
> [ ["Value", "aa"], ["", null] ]

$ aa;;
> [ ["Value","aa"], ["",null], ["",null] ]

$ a;b;c
> [ ["Value","a"], ["b",null], ["c",null] ]

$ a;b;c;
> [ ["Value","a"], ["b",null], ["c",null], ["",null] ]

$ a=1;b=2;c=3
> [ ["a","1"], ["b","2"], ["c","3"] ]

$ 1;b=2;c=3
> [ ["Value","1"], ["b","2"], ["c","3"] ]

// empty values: trying escape sequences using initial tests

$ \;
> [ ["Value",";"] ]

$ \;\;
> [ ["Value",";;"] ]

$ \;\;\;
> [ ["Value",";;;"] ]

$ ;\;
> [ ["Value",""], [";",null] ]

$ ;\;\;
> [ ["Value",""], [";;",null] ]

$ ;\;\;\;
> [ ["Value",""], [";;;",null] ]

$ ;\;\;\;;
> [ ["Value",""], [";;;",null], ["",null] ]

$ ;\;\;\;;;
> [ ["Value",""], [";;;",null], ["",null], ["",null] ]

$ ;\;\;\;;;\;
> [ ["Value",""], [";;;",null], ["",null], [";",null] ]

$ ;\;\;\;;;\;=a
> [ ["Value",""], [";;;",null], ["",null], [";","a"] ]

$ ;\;\;\;;;\;=a\;
> [ ["Value",""], [";;;",null], ["",null], [";","a;"] ]

$ ;\;\;\;;;\;=\;a
> [ ["Value",""], [";;;",null], ["",null], [";",";a"] ]

$ ;\;\;\;;;\;=\;a\;
> [ ["Value",""], [";;;",null], ["",null], [";",";a;"] ]

$ ;\;\;\;;;\;=\;a\;;
> [ ["Value",""], [";;;",null], ["",null], [";",";a;"], ["",null] ]

$ a;;a
> [ ["Value","a"], ["",null], ["a",null] ]

$ a;;\;=a
> [ ["Value","a"], ["",null], [";","a"] ]

$ a;;;\;
> [ ["Value","a"], ["",null], ["",null], [";",null] ]
